16. Contoh-1: animasi lingkaran yang bergerak, pada program ini sebuah lingkaran warna merah dibentuk (digambar) pada posisi berbeda namun berdekatan, dengan latar belakang warna putih dan hitam sehingga efek bergerak dari kiri ke kanan dan sebaliknya, tampak jelas. animasi.java Tolong bikinkan codingannya


Jawaban:

import javax.swing.*;

import java.awt.*;

public class Animasi extends JPanel {

private int x = 0; // Posisi awal lingkaran

public Animasi() {

setBackground(Color.WHITE);

Timer timer = new Timer(10, e -> {

x += 2; // Posisi lingkaran berubah

repaint(); // Menggambar ulang panel

});

timer.start();

}

@Override

protected void paintComponent(Graphics g) {

super.paintComponent(g);

g.setColor(Color.RED);

g.fillOval(x, getHeight() / 2 - 25, 50, 50); // Menggambar lingkaran

}

public static void main(String[] args) {

JFrame frame = new JFrame("Animasi Lingkaran");

frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);

frame.setSize(800, 300);

frame.add(new Animasi());

frame.setVisible(true);

}

}
