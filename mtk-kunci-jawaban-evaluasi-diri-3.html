[tex]\frac{a}{b} + \frac{c}{b} = \frac{a \: + \: c}{b} [/tex]

[tex]a\frac{b}{c} = \frac{a \times c \: + \: b}{c}[/tex][tex]a\frac{b}{c} = a + \frac{b}{c}[/tex]
