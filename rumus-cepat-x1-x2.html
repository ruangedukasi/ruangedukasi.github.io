[tex]ax^2+bx+c=0[/tex]

[tex]\displaystyle x_1^2x_2+x_1x_2^2=x_1x_2(x_1+x_2) \\ x_1^2x_2+x_1x_2^2=\frac{c}{a}(-\frac{b}{a}) \\ x_1^2x_2+x_1x_2^2=-\frac{bc}{a^2}[/tex]
x₁³x₂ + x₁.x₂³ = x₁x₂(x₁² + x₂²)
= x₁x₂((x₁ + x₂)² - 2x₁x₂)
= (c/a)((-b/a)² - 2(c/a)
= (c/a)(b²/a² - 2ac/a²)
= (c/a)((b² - 2ac)/a²)
